Protobuf编译器

Protobuf编译器

下载地址

安装配置

$ sudo chown -R aidan:staff /opt/dev
$ cd /opt/dev
$ mkdir -p /opt/dev/Proto
$ cd /opt/dev/Proto
$ wget https://github.com/protocolbuffers/protobuf/releases/download/v3.14.0/protoc-3.14.0-osx-x86_64.zip
$ unzip protoc-3.14.0-osx-x86_64.zip
$ rm -rf protoc-3.14.0-osx-x86_64.zip
$ vim ~/.profile
export PROTO_HOME=/opt/dev/Proto
export PATH=$PROTO_HOME/bin:$PATH
$ . ~/.profile

$ protoc --version
libprotoc 3.14.0
